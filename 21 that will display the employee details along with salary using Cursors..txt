SET SERVEROUTPUT ON;
DECLARE
    -- Declaring the cursor to fetch employee details
    CURSOR emp_cursor IS 
        SELECT emp_no, name, designation, salary FROM employee;
    
    v_emp_rec emp_cursor%ROWTYPE;
BEGIN
    OPEN emp_cursor;
    DBMS_OUTPUT.PUT_LINE('EMP_NO | NAME | DESIGNATION | SALARY');
    DBMS_OUTPUT.PUT_LINE('---------------------------------------');
    LOOP
        FETCH emp_cursor INTO v_emp_rec;
        EXIT WHEN emp_cursor%NOTFOUND;
        
        DBMS_OUTPUT.PUT_LINE(v_emp_rec.emp_no || ' | ' || v_emp_rec.name || ' | ' || 
                             v_emp_rec.designation || ' | ' || v_emp_rec.salary);
    END LOOP;
    CLOSE emp_cursor;
END;
/
