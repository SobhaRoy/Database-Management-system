CREATE OR REPLACE FUNCTION calculate_age(p_emp_id IN NUMBER) RETURN NUMBER IS
    v_dob DATE;
    v_age NUMBER;
BEGIN
    SELECT dob INTO v_dob FROM employee WHERE emp_no = p_emp_id;
    v_age := FLOOR(MONTHS_BETWEEN(SYSDATE, v_dob) / 12);
    RETURN v_age;
END;
/

DECLARE
    v_age NUMBER;
BEGIN
    v_age := calculate_age(101);
    DBMS_OUTPUT.PUT_LINE('The Age of Employee is: ' || v_age);
END;
/
