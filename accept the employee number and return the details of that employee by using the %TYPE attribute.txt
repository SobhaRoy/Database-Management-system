CREATE OR REPLACE PROCEDURE EMP_INFO(V_EMPID IN EMP.EMPNO%TYPE) IS
	V_EMPNO  EMP.EMPNO%TYPE;
	V_ENAME  EMP.ENAME%TYPE;
	V_JOB EMP.JOB%TYPE;
	V_MGR  EMP.MGR%TYPE;
	V_HIREDATE  EMP.HIREDATE%TYPE;
BEGIN
	SELECT EMPNO,ENAME,JOB,MGR,HIREDATE INTO V_EMPNO,V_ENAME,V_JOB,V_MGR,V_HIREDATE FROM EMP WHERE 	EMPNO=V_EMPID;
	DBMS_OUTPUT.PUT_LINE(V_EMPNO||V_ENAME||V_JOB||V_MGR||V_HIREDATE );
END;
/

SET SERVEROUTPUT ON;
BEGIN
	EMP_INFO(7369);
END;
